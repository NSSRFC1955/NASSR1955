        function moveLight(e) {

        function fireButton(e) {

            e.preventDefault();

            const btn = e.target.closest('.btn');

            btn.classList.add('fire');

            setTimeout(() => {

                btn.classList.remove('fire');

                document.getElementById('player').style.display = 'block';

        }
